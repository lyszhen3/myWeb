# 基础配置
~~~~
" 去除提示音
set vb t_vb=
au GuiEnter * set t_vb=

set guifont=courier_new:h11
set nu
" 忽略大小写
set ic
set enc=UTF-8
set langmenu=zh_CN.UTF-8
source $VIMRUNTIME/delmenu.vim
source $VIMRUNTIME/menu.vim

set noundofile
set nobackup
set noswapfile
set tabstop=4
set shiftwidth=4

vnoremap <C-C> "+y
vnoremap <C-X> "+x
noremap <C-Q> <C-V>
noremap <C-V> "+gp
cmap <C-V> <C-R>+

set fileencodings=ucs-bom,utf-8,cp936,gb18030,big5,euc-jp,euc-kr,latin1
set encoding=utf-8
set termencoding=cp936
language messages zh_CN.UTF-8

" 使用 :Jsonf json格式化
command! Jsonf :execute '%!python -m json.tool --no-ensure-ascii'
~~~~
